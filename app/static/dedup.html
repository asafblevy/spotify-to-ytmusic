YouTube Music Dedup

Remove duplicate songs from a playlist

1 Open your playlist

Go to YouTube Music and open the playlist you want to deduplicate.

2 Open browser console

Press F12 (or Cmd+Option+J on Mac), then click the Console tab.

3 Paste and run this script

It scrolls to load all songs, finds duplicates by title+artist, and removes them via the menu.

async function dedupPlaylist() {
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  // Step 1: Scroll to load all songs
  console.log('Scrolling to load all songs...');
  let last = 0;
  while (true) {
    window.scrollTo(0, document.body.scrollHeight);
    await sleep(1500);
    const current = document.querySelectorAll(
      'ytmusic-responsive-list-item-renderer'
    ).length;
    console.log('Loaded ' + current + ' songs...');
    if (current === last) break;
    last = current;
  }

  const rows = [...document.querySelectorAll(
    'ytmusic-responsive-list-item-renderer'
  )];
  console.log('Total songs: ' + rows.length);

  // Step 2: Find duplicates
  const seen = new Set();
  const dupes = [];
  for (const row of rows) {
    const title = row.querySelector(
      '.title-column yt-formatted-string'
    )?.textContent?.trim() || '';
    const artist = row.querySelector(
      '.secondary-flex-columns yt-formatted-string'
    )?.textContent?.trim() || '';
    const key = title.toLowerCase() + '|||' + artist.toLowerCase();
    if (seen.has(key)) {
      dupes.push({ row, title, artist });
    } else {
      seen.add(key);
    }
  }

  console.log('Found ' + dupes.length + ' duplicates');
  if (dupes.length === 0) {
    console.log('No duplicates found!');
    return;
  }

  // Step 3: Remove duplicates (from bottom to top to avoid index shifting)
  dupes.reverse();
  let removed = 0;
  for (const dupe of dupes) {
    // Scroll the row into view
    dupe.row.scrollIntoView({ block: 'center' });
    await sleep(300);

    // Click the 3-dot menu
    const menuBtn = dupe.row.querySelector(
      'ytmusic-menu-renderer yt-button-shape button'
    );
    if (!menuBtn) { console.log('No menu for: ' + dupe.title); continue; }
    menuBtn.click();
    await sleep(400);

    // Find and click "Remove from playlist"
    const menuItems = document.querySelectorAll(
      'tp-yt-paper-listbox ytmusic-menu-service-item-renderer'
    );
    let found = false;
    for (const item of menuItems) {
      const text = item.textContent.toLowerCase();
      if (text.includes('remove from playlist') ||
          text.includes('הסרה מהפלייליסט') ||
          text.includes('הסר')) {
        item.click();
        removed++;
        console.log('Removed (' + removed + '/' + dupes.length +
          '): ' + dupe.title + ' - ' + dupe.artist);
        found = true;
        await sleep(500);
        break;
      }
    }
    if (!found) {
      // Close menu if remove wasn't found
      document.body.click();
      await sleep(200);
      console.log('Could not find remove option for: ' + dupe.title);
    }
  }
  console.log('Done! Removed ' + removed + ' duplicates.');
}
dedupPlaylist();

4 Wait

Keep the tab open. It removes one duplicate at a time. Progress is logged in the console.

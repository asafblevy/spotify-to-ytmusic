Bulk Like Songs

Like every song in a YouTube Music playlist

1 Open your playlist

Go to YouTube Music and open the playlist you want to like all songs from.

2 Open browser console

Desktop: Press F12 (or Cmd+Option+J on Mac), then click the Console tab.
Mobile: This only works on desktop browsers.

3 Paste and run this script

It will scroll to load all songs, then like each one with a short delay.

async function likeAll() {
  // Step 1: Scroll to load all songs
  console.log('Scrolling to load all songs...');
  let last = 0;
  while (true) {
    window.scrollTo(0, document.body.scrollHeight);
    await new Promise(r => setTimeout(r, 1500));
    const current = document.querySelectorAll(
      'ytmusic-responsive-list-item-renderer'
    ).length;
    console.log('Loaded ' + current + ' songs...');
    if (current === last) break;
    last = current;
  }

  // Step 2: Find all unliked songs
  const buttons = document.querySelectorAll(
    'yt-button-shape#button-shape-like button[aria-pressed="false"]'
  );
  console.log('Found ' + buttons.length + ' unliked songs');

  // Step 3: Like each one
  for (let i = 0; i < buttons.length; i++) {
    buttons[i].click();
    console.log('Liked ' + (i + 1) + '/' + buttons.length);
    await new Promise(r => setTimeout(r, 500));
  }
  console.log('Done! Liked ' + buttons.length + ' songs');
}
likeAll();

4 Wait

Keep the tab open. It takes ~12 minutes per 1400 songs. Progress is logged in the console.
